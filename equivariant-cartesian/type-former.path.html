--‚Üì A path is a function from the interval shape with specified endpoints.

record _~_ {A : Type ‚Ñì} (a‚ÇÄ a‚ÇÅ : A) : Type ‚Ñì where
  constructor path
  field

    at0 : at 0 ‚â° a‚ÇÄ

eqToPath : {A : Type ‚Ñì} {a‚ÇÄ a‚ÇÅ : A} ‚Üí a‚ÇÄ ‚â° a‚ÇÅ ‚Üí a‚ÇÄ ~ a‚ÇÅ

--‚Üì The reflexive path is the constant function.

refl~ : {A : Type ‚Ñì} (a : A) ‚Üí a ~ a

congPath : {A : Type ‚Ñì} {B : Type ‚Ñì'} (f : A ‚Üí B) {a‚ÇÄ a‚ÇÅ : A}
  ‚Üí a‚ÇÄ ~ a‚ÇÅ ‚Üí f a‚ÇÄ ~ f a‚ÇÅ
congPath f p .at = f ‚àò p .at
congPath f p .at0 = cong f (p .at0)
congPath f p .at1 = cong f (p .at1)

--‚Üì An extensionality principle for paths.

PathExt : {A : Type ‚Ñì} {a‚ÇÄ a‚ÇÅ : A} {p q : a‚ÇÄ ~ a‚ÇÅ}
  ‚Üí (‚àÄ i ‚Üí p .at i ‚â° q .at i) ‚Üí p ‚â° q
PathExt t =
  congŒ£ (uncurry ‚àò path) (funExt t) (√óext uip' uip')

--‚Üì Definition of the family underlying the path type.

congPathÀ£ : {A : Œì ‚Üí Type ‚Ñì} {B : Œì ‚Üí Type ‚Ñì'}
  (f : Œì ‚ä¢À£ A ‚ÜíÀ£ B)
  {a‚ÇÄ a‚ÇÅ : Œì ‚ä¢À£ A} (p : Œì ‚ä¢À£ PathÀ£ A a‚ÇÄ a‚ÇÅ)
  ‚Üí Œì ‚ä¢À£ PathÀ£ B (appÀ£ f a‚ÇÄ) (appÀ£ f a‚ÇÅ)
congPathÀ£ f p Œ≥ = congPath (f Œ≥) (p Œ≥)

opaque
  private
    partialEl : {A : Œì ‚Üí Type ‚Ñì} (a‚ÇÄ a‚ÇÅ : Œì ‚ä¢À£ A)
      ‚Üí Œì ‚ñ∑ùïÄ ‚ñ∑[ ‚àÇ ‚àò ùíí ] ‚ä¢À£ (A ‚àò ùíë) ‚Üæ (‚àÇ ‚àò ùíí)
    partialEl a‚ÇÄ a‚ÇÅ =
      uncurry Œª (Œ≥ , i) ‚Üí ‚àÇ-rec i (Œª _ ‚Üí a‚ÇÄ Œ≥) (Œª _ ‚Üí a‚ÇÅ Œ≥)

    retract : {A : Œì ‚Üí Type ‚Ñì} {a‚ÇÄ a‚ÇÅ : Œì ‚ä¢À£ A}
      ‚Üí Œì ‚ä¢À£ RetractÀ£ (PathÀ£ A a‚ÇÄ a‚ÇÅ) (ExtensionÀ£ ùïö (A ‚àò ùíë) ‚àÇ (partialEl a‚ÇÄ a‚ÇÅ))

    retract Œ≥ .sec p i .out‚â° =
      ‚àÇ-elim i (Œª {refl ‚Üí sym (p .at0)}) (Œª {refl ‚Üí sym (p .at1)})
    retract Œ≥ .ret ex .at i = ex i .out

    retract Œ≥ .ret ex .at1 = sym (ex 1 .out‚â° (‚à®r refl))
    retract Œ≥ .inv _ = PathExt Œª _ ‚Üí refl

  PathFibStr : {A : Œì ‚Üí Type ‚Ñì} (Œ± : FibStr A) (a‚ÇÄ a‚ÇÅ : Œì ‚ä¢À£ A)
    ‚Üí FibStr (PathÀ£ A a‚ÇÄ a‚ÇÅ)

    retractFibStr retract (ExtensionFibStr ùïö (Œ± ‚àò·∂†À¢ ùíë) ‚àÇ _)

  --‚Üì The fibrancy structure is stable under reindexing.

  reindexPathFibStr : {A : Œì ‚Üí Type ‚Ñì} {Œ± : FibStr A} {a‚ÇÄ a‚ÇÅ : Œì ‚ä¢À£ A}
    (œÅ : Œî ‚Üí Œì)
    ‚Üí PathFibStr Œ± a‚ÇÄ a‚ÇÅ ‚àò·∂†À¢ œÅ ‚â° PathFibStr (Œ± ‚àò·∂†À¢ œÅ) (a‚ÇÄ ‚àò œÅ) (a‚ÇÅ ‚àò œÅ)

    reindexRetractFibStr retract œÅ
    ‚àô
    cong‚ÇÇ
      retractFibStr
      (funExt' $ retractExt (funExt' $ funExt' $ restrictExt refl) refl)
      (reindexExtensionFibStr œÅ)

Fiber : {A : Type ‚Ñì} {B : Type ‚Ñì'} (f : A ‚Üí B) (b : B) ‚Üí Type (‚Ñì ‚äî ‚Ñì')
Fiber f b = Œ£ a ‚àà _ , f a ~ b

FiberÀ£ : {A : Œì ‚Üí Type ‚Ñì} {B : Œì ‚Üí Type ‚Ñì'} (f : Œì ‚ä¢À£ A ‚ÜíÀ£ B) (b : Œì ‚ä¢À£ B)
  ‚Üí (Œì ‚Üí Type (‚Ñì ‚äî ‚Ñì'))
FiberÀ£ f b Œ≥ = Fiber (f Œ≥) (b Œ≥)

opaque
  FiberFibStr : {A : Œì ‚Üí Type ‚Ñì} (Œ± : FibStr A) {B : Œì ‚Üí Type ‚Ñì'} (Œ≤ : FibStr B)
    (f : Œì ‚ä¢À£ A ‚ÜíÀ£ B) (b : Œì ‚ä¢À£ B)
    ‚Üí FibStr (FiberÀ£ f b)
  FiberFibStr Œ± Œ≤ f b =
    Œ£FibStr Œ± (PathFibStr (Œ≤ ‚àò·∂†À¢ ùíë) (uncurry f) (b ‚àò ùíë))

  reindexFiberFibStr : {A : Œì ‚Üí Type ‚Ñì} {Œ± : FibStr A} {B : Œì ‚Üí Type ‚Ñì'} {Œ≤ : FibStr B}
    {f : Œì ‚ä¢À£ A ‚ÜíÀ£ B} {b : Œì ‚ä¢À£ B}
    (œÅ : Œî ‚Üí Œì)
    ‚Üí FiberFibStr Œ± Œ≤ f b ‚àò·∂†À¢ œÅ ‚â° FiberFibStr (Œ± ‚àò·∂†À¢ œÅ) (Œ≤ ‚àò·∂†À¢ œÅ) (f ‚àò œÅ) (b ‚àò œÅ)

    reindexŒ£FibStr _ ‚àô cong (Œ£FibStr _) (reindexPathFibStr _)

Fiber·∂† : (A : Œì ‚ä¢·∂†Type ‚Ñì) (B : Œì ‚ä¢·∂†Type ‚Ñì') (f : Œì ‚ä¢·∂† A ‚Üí·∂† B) (b : Œì ‚ä¢·∂† B)
  ‚Üí Œì ‚ä¢·∂†Type (‚Ñì ‚äî ‚Ñì')

Fiber·∂† A B f b .snd = FiberFibStr (A .snd) (B .snd) f b

Singl : (A : Type ‚Ñì) (a : A) ‚Üí Type ‚Ñì
Singl A = Fiber id

Singl·∂† : (A : Œì ‚ä¢·∂†Type ‚Ñì) (a : Œì ‚ä¢·∂† A) ‚Üí Œì ‚ä¢·∂†Type ‚Ñì

singlCenter : {A : Type ‚Ñì} (a : A) ‚Üí Singl A a

singlCenter·∂† : (A : Œì ‚ä¢·∂†Type ‚Ñì) (a : Œì ‚ä¢·∂† A)
  ‚Üí Œì ‚ä¢·∂† Singl·∂† A a
singlCenter·∂† A a =
  pair·∂† A (Path·∂† (A ‚àò·∂† ùíë) ùíí (a ‚àò ùíë)) a (refl·∂† A a)

opaque
  singlContract : (A : ùüô ‚ä¢·∂†Type ‚Ñì) (a : A $·∂† tt) (c : Singl·∂† A (cst a) $·∂† tt)
    ‚Üí singlCenter a ~ c
  singlContract A a c = homotopy
    where
    box : (i : ùïÄ) ‚Üí OpenBox ùïö (‚à£ A ‚à£ ‚àò cst tt) 1
    box i .cof = ‚àÇ i
    box i .tube j = ‚àÇ-rec i (Œª {refl ‚Üí a}) (Œª {refl ‚Üí c .snd .at j})
    box i .cap .out = a
    box i .cap .out‚â° = ‚àÇ-elim i (Œª {refl ‚Üí refl}) (Œª {refl ‚Üí c .snd .at1})

    square : (i : ùïÄ) ‚Üí Filler (box i)
    square i = A .snd .lift ùïö _ 1 (box i)

    homotopy : (a , refl~ a) ~ c
    homotopy .at i .fst = square i .fill 0 .out
    homotopy .at i .snd = path (Œª j ‚Üí square i .fill j .out) refl (square i .cap‚â°)
    homotopy .at0 =
      FiberExt
        (sym (square 0 .fill 0 .out‚â° (‚à®l refl)))
        (Œª j ‚Üí sym (square 0 .fill j .out‚â° (‚à®l refl)))
    homotopy .at1 =
      FiberExt
        (sym (square 1 .fill 0 .out‚â° (‚à®r refl)) ‚àô c .snd .at0)
        (Œª j ‚Üí sym (square 1 .fill j .out‚â° (‚à®r refl)))

singlContract·∂† : (A : Œì ‚ä¢·∂†Type ‚Ñì) (a : Œì ‚ä¢·∂† A) (c : Œì ‚ä¢·∂† Singl·∂† A a)
  ‚Üí Œì ‚ä¢·∂† Path·∂† (Singl·∂† A a) (singlCenter·∂† A a) c
singlContract·∂† A a c Œ≥ = singlContract (A ‚àò·∂† cst Œ≥) (a Œ≥) (c Œ≥)

subst·∂† : (A : Œì ‚ä¢·∂†Type ‚Ñì) (B : Œì ‚ñ∑·∂† A ‚ä¢·∂†Type ‚Ñì') {a‚ÇÄ a‚ÇÅ : Œì ‚ä¢·∂† A}
  (p : Œì ‚ä¢·∂† Path·∂† A a‚ÇÄ a‚ÇÅ)
  ‚Üí Œì ‚ä¢·∂† B ‚àò·∂† (id ,, a‚ÇÄ)
  ‚Üí Œì ‚ä¢·∂† B ‚àò·∂† (id ,, a‚ÇÅ)
subst·∂† A B p b‚ÇÄ Œ≥ =
  subst (‚à£ B ‚à£ ‚àò (Œ≥ ,_)) (p Œ≥ .at1)
    (fibTranspStr (B ‚àò·∂† (cst Œ≥ ,, p Œ≥ .at)) .lift ùïö id
      0
      (subst (‚à£ B ‚à£ ‚àò (Œ≥ ,_)) (sym (p Œ≥ .at0)) (b‚ÇÄ Œ≥))
      1)

J·∂† : (A : Œì ‚ä¢·∂†Type ‚Ñì) (a : Œì ‚ä¢·∂† A)

  subst·∂† (Singl·∂† A a) P (singlContract·∂† A a c) d
